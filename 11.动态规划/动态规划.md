## 图基础
顶点vertex
边edge
存储方式:邻接表、邻接矩阵等

建议方式
```c++
class graph
{
public:
    graph()
    {
    }
    unordered_map<int,*vertex> vertexs;
    unordered_set<*edge> edges;
};

template<class T>
class vertex
{
public:
    vertex( T value_):from(0),to(0);
    {
        value = value_;

    }
    T value;
    int to;//入度
    int from;//出度
    vector<edge*> edges;
    vector<vertex*> nexts;
};

class edge
{
public:
    edge(int distance_,vertex * from_,vertex * to_)
    {
        distance = distance_;
        from = from_;
        to = to_;
    }
    
    vertex *from;
    vertex *to;
    int distance;
};
```
